# 单例模式

https://github.com/ccb1900/pages/tree/master/example/go/singleton (opens new window)

# 普通实现（不支持并发）

package general

type Singleton struct {
}

var instance *Singleton

func GetInstance() *Singleton {
	if instance == nil {
		instance = new(Singleton)
	}

	return instance
}

# 加锁实现

package mutex

import "sync"

type Singleton struct {
}

var instance *Singleton
var lock sync.Mutex

func GetInstance() *Singleton {
	lock.Lock()
	if instance == nil {
		instance = new(Singleton)
	}
	lock.Unlock()

	return instance
}

# sync.once 实现

package special

import (
	"fmt"
	"sync"
)

type Singleton struct {
}

var instance *Singleton
var once sync.Once

func GetInstance() *Singleton {
	// 仅执行一次
	once.Do(func() {
		fmt.Println("once")
		instance = new(Singleton)
	})

	return instance
}
